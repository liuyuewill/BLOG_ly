```JS
        return this.price * this.amount
    }
```

```js
computed:

@computed

autorun:
var numbers = observable([1,2,3]);
var sum = computed(() => numbers.reduce((a, b) => a + b, 0));
var disposer = autorun(() => console.log(sum.get())); // 输出 '6' // 自己会立即执行一次
numbers.push(4); // 输出 '10'
disposer(); // !!!!!!!!!!!
```

### action

唯一需要注意的是，在严格模式下，对于异步action里的回调，若该回调也要修改observable的值，那么该回调也需要绑定action。

```JS
const Mobx = require("mobx");
Mobx.configure({ enforceActions: true });
const { observable, autorun, computed, extendObservable, action } = Mobx;
class Store {
  @observable a = 123;

  @action
  changeA() {
    this.a = 0;
    setTimeout(this.changeB, 1000);
  }
  @action.bound // 回调也要绑定@action. 这里用了action.bound语法糖，目的是为了解决javascript作用域问题
  changeB() {
    this.a = 1000;
  }
}
var s = new Store();
autorun(() => console.log(s.a));
s.changeA();


// 更简单的写法
const Mobx = require("mobx");
Mobx.configure({ enforceActions: true });
const { observable, autorun, computed, extendObservable, action } = Mobx;
class Store {
  @observable a = 123;
  @action
  changeA() {
    this.a = 0;
    setTimeout(action('changeB',()=>{
      this.a = 1000;
    }), 1000);
  }
}
var s = new Store();
autorun(() => console.log(s.a));
s.changeA();


// 另一种写法 runInAction
 @action
  changeA() {
    this.a = 0;
    setTimeout(
      runInAction(() => {
        this.a = 1000;
      }),
      1000
    );
  }
```

### when

```JS
class Store {
  @observable a = 1;
  constructor() {
    when(
      () => {
        return this.a > 10;
      },
      function() {
        console.log("good");
      }
    );
  }
}
var s = new Store();
s.a=3
s.a=5
s.a = 7
s.a=11 // 就会打印出 good
```
